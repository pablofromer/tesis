esMujer(?y), esMujer(?x), sonPareja(?x, ?y) -> lesbiana(?x).
sonPareja(?y, ?x) -> sonPareja(?x, ?y).
esHombre(?y), esMujer(?x), sonPareja(?x, ?y) -> hetero(?x).
lesbiana(?x), hetero(?x) -> bisexual(?x).
